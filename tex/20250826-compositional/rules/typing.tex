% Expression Typing

\newcommand*\RuleTConst{
  \inferrule[T-Const]{
    \HasCType
      \C
      \T
  }{
    \HasType[\EffPure]
      \CtxEmpty
      \C
      \T
  }
}

\newcommand*\RuleTVar{
  \inferrule[T-Var]{}{
    \HasType[\EffPure]
      {\CtxVar \EVar\T}
      \EVar
      \T
  }
}

\newcommand*\RuleTFork{
  \inferrule[T-Fork]{
    \Bounded \Ctx
    \\
    \HasType[\Eff[1]]
      {\Ctx}
      {\E}
      {\TArr{\MobMobile}{\Eff} {\BUnit} {\BUnit}}
  }{
    \HasType[\EffPure]
      \Ctx
      {\EFork \E}
      {\BUnit}
  }
}

}

\newcommand*\RuleTAppUnr{
  \inferrule[T-AppUnr]{
    \HasType[\Eff[1]]
      {\Ctx_1}
      {\E[1]}

    \HasType[\Eff[2]]
      {\Ctx_2}
      {\E[2]}
      {\T[1]}
  }{
    \HasType[\Eff[1] \EffJoin \Eff[2] \EffJoin \Eff[3]]
      {\CtxSeq {\Ctx_1}{\Ctx_2}}
      {(\EApp{\E[1]}{\E[2]})}

  }
}

\newcommand*\RuleTAppLin{

  }
}

\newcommand*\RuleTAppLeft{
  \inferrule[T-AppLeft]{
    % \Unr{\Ctx_1} \vee \Eff[1] = \EffPure
    % \\
    \HasType
      {\Ctx_1}
      {\E[1]}
      {(\TArr{\Mob}{\DirLeft,{\Eff[3]}}{\T[1]}{\T[2]})}
      {\EffPure} \\
    \HasType
      {\Ctx_2}
      {\E[2]}
      {\T[1]}
      {\Eff[2]}
  }{
    \HasType
      {\CtxSeq {\Ctx_2}{\Ctx_1}}
      {(\EApp{\E[1]}{\E[2]})}
      {\T[2]}
      {(\Eff[2] \EffJoin \Eff[3])}
  }
}

\newcommand*\RuleTAppRight{
  \inferrule[T-AppRight]{

    % \\
    \HasType

      {\Eff[1]} \\
    \HasType
      {\Ctx_2}
      {\E[2]}
      {\T[1]}
      {\EffPure}
  }{
    \HasType
      {\CtxSeq {\Ctx_1}{\Ctx_2}}
      {(\EApp{\E[1]}{\E[2]})}
      {\T[2]}
      {(\Eff[1] \EffJoin \Eff[3])}
  }
}

\newcommand*\RuleTPairUnord{
  \inferrule[T-PairUnord]{
    \HasType{\Eff[1]}
      {\Ctx_1}
      {\E[1]}
      {\T[1]} \\
    \HasType[\Eff[2]]
      {\Ctx_2}
      {\E[2]}
      {\T[2]}
  }{
    \HasType[\Eff[1] \EffJoin \Eff[2]]
      {\CtxPar {\Ctx_1}{\Ctx_2}}
      {\EPair{\E[1]}{\E[2]}}
      {\TPairUnord{\T[1]}{\T[2]}}
  }
}

\newcommand*\RuleTPairOrd{
  \inferrule[T-PairOrd]{
    % \Unr{\Ctx_1} \vee \Unr {\T[1]} \vee \Eff[2] = \EffPure \\
    \HasType[\Eff[1]]
      {\Ctx_1}
      {\E[1]}
      {\T[1]} \\
    \HasType[\EffPure]
      {\Ctx_2}
      {\E[2]}
      {\T[2]}
  }{
    \HasType[\Eff[1]]
      {\CtxSeq {\Ctx_1}{\Ctx_2}}
      {\EPair{\E[1]}{\E[2]}}
      {\TPairOrd{\T[1]}{\T[2]}}
  }
}

\newcommand*\RuleTPack{
  \inferrule[T-Pack]{
    \HasTargetType
      \Ctx
      {\E}
      {\T{[{\S}/\alpha]}}
  }{
    \HasTargetType
      \Ctx
      {\EPack{\S}{\E}{\TExists[\pr{S}]\alpha\T}}
      {\TExists[\pr{S}]\alpha\T}
  }
}

\newcommand*\RuleTUnpack{
  \inferrule[T-Unpack]{
    \HasTargetType
      {\Ctx}
      {\E[1]}
      {\TExists\alpha{\T[1]}}
    \\
    \HasTargetType
      {\CtxExtTarget{\CtxExtTarget\Ctx\alpha}{\CtxVar x {\T[1]}}}
      {\E[2]}
      {\T[2]}
    \\
    \alpha \notin \T[2]
  }{
    \HasTargetType
      \Ctx
      {\EOpen\alpha x {\E[1]} {\E[2]}}
      {\T[2]}
  }
}
