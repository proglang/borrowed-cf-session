% CS-Pair
\DefRule {Pair} {} {
  \AT \TPair \ATU \CEq \AT* \TPair \ATU*
  \csred
  \left\{
    \AT \CEq \AT*\!,
    \ATU \CEq \ATU*
  \right\}
}

\DefRule {Sum} {} {
  \TSum {\AT} {\ATU} \CEq \TSum {\AT*} {\ATU*}
  \csred
  \left\{
    \AT \CEq \AT*\!,
    \ATU \CEq \ATU*
  \right\}
}

\DefRule {Unit} {} {
  \BUnit \CEq \BUnit
  \csred
  \emptyset
}

\DefRule {Arrow} {} {
  \TArr{\Mob}{\Dir,\Eff}{\AT}{\ATU}
  \CEq
  \TArr{\Mob}{\Dir,\Eff}{\AT*}{\ATU*}
  \csred
  \{
    \AT* \CEq \AT\!,
    \ATU \CEq \ATU*
  \}
}

\DefRule {Ret} {} {
  \CBounded \SRet \csred \emptyset
}

\DefRule {Close} {} {
  \CBounded \STerm \csred \emptyset
}

\DefRule {Wait} {} {
  \CBounded \SWait \csred \emptyset
}

}{

}

\DefRule {SeqII} {

}{
  \CBounded\ (\SSeq{\AS[1]}{\AS[2]}) \csred \{ \CBounded \AS[2] \}
}
