% Reduction rules

\def\RuleCmdPrefix{RuleUProcRed}
\def\RuleNamePrefix{RU-}

% RU-Exp
\DefRule {Exp} {
  \E[1]  \EReduce \E[2]
}{
  \TFCapp{\E[1]} \TPReduce \TFCapp{\E[2]}
}

% RU-Fork
\DefRule {Fork} {} {
  \TFCapp{\EApp \CFork \V}
  \TPReduce
  \TPPar{\TFCapp\CUnit}{\TPExp{\EApp\V\CUnit}}
}

% RU-New
\DefRule {New} {} {
  \TFCapp{\EApp \CNew \CUnit}
  \TPReduce
  \TPNew {
    \TPSync [z_1] \FlagDropped
    \TPSync [z_2] \FlagDropped
    (
      \TFCapp{\EPair
        {(\TEChan [\TEVarz[1]] \TEVar)}
        {(\TEChan [\TEVarz[2]] \TEVary)}
      }
    )
  }
}

% RU-LSplit
\DefRule {LSplit} {} {
    \TPNew {(
      \TPPar {\TFCapp{\EApp \CSplit {(\TEChan[\E[1]]{\TEVar}[\E[2]])}}} \TProc
    )}
    \TPReduce
    \TPNew {(
      \TPPar
        {\TFCapp{\EPair{(\TEChan[\E[1]]{\TEVar})}{(\TEChan{\TEVar}[\E[2]])}}}
        \TProc
    )}
}

% RU-RSplit
\DefRule {RSplit} {} {
  \TPNew {(
    \TPPar {\TFCapp{\EApp \CSSplit {(\TEChan [e_1] \TEVar [e_2])}}} \TProc
  )}
  \TPReduce
  \TPNew {
    \TPSync \FlagInit (
      \TPPar {
        \TFCapp{\EPair
          {(\TEChan [e_1]     \TEVar [\TEVarz])}
          {(\TEChan [\TEVarz] \TEVar [e_2])}
        }
      } \TProc
    )
  }
}

% RU-Cleanup
\DefRule {Cleanup} {} {
  \TPSync \FlagAcquired \TProc
  \TPReduce
  \Subs \TProc \TEVarz \CUnit
}

% RU-Drop
\DefRule {Drop} {} {
    \TPNew {\TPSync \FlagInit (
      \TPPar {\TFCapp{\EApp \CDrop {(\TEChan [\E] \TEVar [\TEVarz])}}} \TProc
    )}
    \TPReduce
    \TPNew {\TPSync \FlagDropped (
      \TPPar {\TFCapp{\CUnit}} \TProc
    )}
}

% RU-Acquire
\DefRule {Acquire} {} {
    \TPNew {
      \TPSync \FlagDropped 
        (\TPPar {\TFCapp{\EApp \CAcq {(\TEChan [\TEVarz] \TEVar [e])}}} \TProc)
    }
    \TPReduce
    \TPNew {
      \TPSync \FlagAcquired 
        (\TPPar {\TFCapp{\TEChan [\TEVarz] \TEVar[e]}} \TProc)
    }
}

% RU-Close
\DefRule {Close} {} {
    \TPNew {(
      \TPPar {\TFCapp[1]{\EApp\CTerm {\ChanPat[1]\TEVar}}}
             {\TFCapp[2]{\EApp\CWait {\ChanPat[2]\TEVary}}}
    )}
    \TPReduce
    \TPPar {\TFCapp[1]{\CUnit}} {\TFCapp[2]{\CUnit}}
}

% RU-Com
\DefRule {Com} {} {
  \TPNew {(
    \TPPar {\TFCapp[1]{\EApp {\CSend} {(\EPair \TV {\ChanPat[1]\TEVar})}}} {
    \TPPar {\TFCapp[2]{\EApp {\CRecv} {\ChanPat[2]\TEVary}}} {\TProc}}
  )}
  \TPReduce
  \TPNew {(
    \TPPar {\TFCapp[1]{\CUnit}} {
    \TPPar {\TFCapp[2]{\TV}} {\TProc}}
  )}
}

% RU-Choice
\DefRule {Choice} {} {
  \TPNew {(
    \TPPar {\TFCapp[1]{\EApp {\CSelect k} {\ChanPat[1] \TEVar}}} {
    \TPPar {\TFCapp[2]{\EApp {\CBranch}   {\ChanPat[2] \TEVary}}} {\TProc}}
  )}
  \TPReduce\\\\
  \TPNew {(
    \TPPar {\TFCapp[1]{\ChanPat[1] \TEVar}} {
    \TPPar {\TFCapp[2]{\EInj k {\ChanPat[2] \TEVary}}} {\TProc}}
  )}
}

% RU-Struct
\DefRule {Struct} {
  \TProc[1] \equiv \TProc*[1] \\
  \TProc*[1] \TPReduce \TProc*[2] \\
  \TProc*[2] \equiv \TProc[2]
}{
  \TProc[1] \TPReduce \TProc[2]
}

% RU-Par
\DefRule {Par} {
  \TProc[1] \TPReduce \TProc[2]
}{
  \TPPar {\TProc[1]}{\TProcQ} \TPReduce \TPPar {\TProc[2]}{\TProcQ}
}

% RU-Bind
\DefRule {Bind} {
  \TProc[1] \TPReduce \TProc[2]
}{
  \TPNew {\TProc[1]} \TPReduce \TPNew {\TProc[2]}
}

% RU-Sync
\DefRule {Sync} {
  \TProc[1] \TPReduce \TProc[2]
}{
  \TPSync \Flag {\TProc[1]} \TPReduce \TPSync \Flag {\TProc[2]}
}
