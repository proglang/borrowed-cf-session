% Algorithmic Expression Typing

\def\RuleNamePrefix{A-}
\def\RuleCmdPrefix{RuleAlg}

% A-Const
\DefRule [(\forall k.~\C \notin \{\CSplit,\CSSplit,\CSelect k,\CBranch\})] {Const} {
  \CtxEmpty \SubCtx \Ctx \\
  \HasCType \C \T
}{
  \InferType \C \T \EffPure [\CSEmpty]
}

% A-LSplit
\DefRule {LSplit} {
  \CtxEmpty \SubCtx \Ctx \\
  \TVar\ \text{fresh}
}{
  \InferType {\CSplit_{\S}}
  {\TArr{\MobMobile}{\DirUnord,\EffPure} {\SSeq\S\TVar} {\TPairOrd{\S}{\TVar}}}
    \EffPure
    [\CSEmpty]
}

% A-RSplit
\DefRule {RSplit} {
  \CtxEmpty \SubCtx \Ctx \\
  \TVar\ \text{fresh}
}{
  \InferType {\CSSplit_{\S}}
  {\TArr{\MobMobile}{\DirUnord,\EffPure} {\SSeq\S\TVar} {\TPairUnord{\SSeq{\S}{\SRet}}{\SSeq\SAcq\TVar}}}
    \EffPure
    [\CSEmpty]
}

% A-Select
\DefRule {Select} {
  \InferType
    \E
    {\SChoice{\ell:\S[\ell]}{\ell\in L}}
    \Eff
}{
  \InferType
    {\EApp{\CSelect k}{\E}}
    {\S[k]}
    {\EffImpure}
}

% A-Branch
\DefRule {Branch} {
  \InferType
    \E
    {\SBranch{\ell:\S[\ell]}{\ell\in L}}
    \Eff
}{
  \InferType
    {\EApp{\CBranch}{\E}}
    {\SVariant{\ell:\S[\ell]}{\ell\in L}}
    {\EffImpure}
}

% A-Var
\DefRule {Var} {
  \CtxVar \EVar \T \SubCtx \Ctx
} {
  \InferType \EVar {\T} \EffPure [\CSEmpty]
}

% A-App
\DefRule {App} {
  \reqDir \Dir \DirPrime \\
  \reqEff\Dir{\Eff[1]}{\Eff[2]} \\
  \CtxAlgJoinCheck{\E[1]}{\E[2]} \\\\
  \InferType [\CtxRestrict{\E[1]}] {\E[1]} {\TArr{\Mob}{\Dir,\Eff[3]} \T \TU} {\Eff[1]} [\CSet[1]] \\
  \CheckType [\CtxRestrict{\E[2]}] {\E[2]} {\T} {\Eff[2]} [\CSet[2]]
}{
  \InferType {\EAppD{\E[1]}{\E[2]}{\DirPrime}} {\TU} {\Eff[1] \EffJoin \Eff[2] \EffJoin \Eff[3]} [\CSet[1] \CSJoin \CSet[2]]
}

% A-Check
\DefRule {Check} {
  \InferType \E {\TU} {\Eff} \\
}{
  \CheckType \E {\T} {\Eff} [\CSet \CSJoin \{\T \CEq \TU\}]
}

% A-Abs
\DefRule {Abs} {
  \CheckType[\CtxVar\EVar\T \CtxJoin \Ctx]
    \E
    \TU
    {\Eff*} \\
  \Eff* \le \Eff \\
  \Dir = \DirNone \implies \Unr \Ctx \wedge \Mob = \MobMobile
}{
  \CheckType
    {\EAbs\EVar\E}
    {\TArr{\Mob}{\Dir,\Eff}{\T}{\TU}}
    {\EffPure}
    [\CSet \cup \left\{ \CMobile {\T*} \mid \CtxVar \EVar {\T*} \in \Ctx \wedge \Mob = \MobMobile \right\}]
}

% A-AbsRec
\DefRule {AbsRec} {
  \CheckType[
    \CtxVar f {\TArr{\MobMobile}{\DirNone,\Eff} \T \TU}
      \CtxPar \CtxVar\EVar\T
      \CtxPar \Ctx
  ]
    \E
    \TU
    {\Eff*} \\
  \Eff* \le \Eff \\
  \Unr\Ctx \\
}{
  \CheckType
    {\EMu {\EAbs\EVar\E}}
    {\TArr{\MobMobile}{\DirNone,\Eff} \T \TU}
    {\EffPure}
}

% A-Pair
\DefRule {Pair} {
  \reqEff \Dir {\Eff[2]} {\Eff[1]}\\
  \CtxAlgJoinCheck{\E[1]}{\E[2]} \\\\
  \CheckType[\CtxRestrict{\E[1]}]
    {\E[1]}
    {\T}
    {\Eff[1]}
    [\CSet[1]]
  \\
  \CheckType[\CtxRestrict{\E[2]}]
    {\E[2]}
    {\TU}
    {\Eff[2]}
    [\CSet[2]]
} {
  \CheckType
    {\EPair {\E[1]} {\E[2]}}
    {\T \TPair \TU}
    {\Eff[1] \EffJoin \Eff[2]}
    [\CSet[1] \CSJoin \CSet[2]]
}

% A-Seq
\DefRule {Seq} {
  \CtxAlgSeqCheck{\E[1]}{\E[2]} \\
  \InferType[\CtxRestrict{\E[1]}]
    {\E[1]}
    {\T}
    {\Eff[1]}
    [\CSet[1]]
  \\
  \InferType[\CtxRestrict{\E[2]}]
    {\E[2]}
    {\TU}
    {\Eff[2]}
    [\CSet[2]]
  \\
  \Unr \T
}{
  \InferType
    {\ESeq{\E[1]}{\E[2]}}
    {\TU}
    {\Eff[1] \EffJoin \Eff[2]}
    [\CSet[1] \CSJoin \CSet[2]]
}

% A-LetPair
\DefRule {LetPair} {
  \CtxAlgSeqCheck{\E[1]}{\E[2]} \\
  \InferType[\CtxRestrict{\E[1]}]
    {\E[1]}
    {\T[1] \TPair \T[2]}
    {\Eff[1]}
    [\CSet[1]]
  \\
  \InferType[(\CtxVar x {\T[1]} \CtxJoin \CtxVar y {\T[2]}) \CtxSeq \CtxRestrict{\E[2]}]
    {\E[2]}
    {\TU}
    {\Eff[2]}
    [\CSet[2]]
}{
  \InferType
     {\ELetPair xy{\E[1]}{\E[2]}}
     {\TU}
     {\Eff[1] \EffJoin \Eff[2]}
     [\CSet[1] \CSJoin \CSet[2]]
}

% A-Inj
\DefRule {Inj} {
  \CheckType
    {\E}
    {\T[\ell]}
    {\Eff}
}{
  \CheckType
    {\EInj k \E}
    {\TVariant{\ell:\T[\ell]}}
    {\Eff}
}

% A-Case
\DefRule {Case} {
  \forall\ell.\CtxAlgSeqCheck{\E}{\E[\ell]}
  \\
  \InferType[\CtxRestrict{\E}]
    {\E}
    {\TVariant{\ell:\T[\ell]}}
    {\Eff}
  \\
  \InferType[\CtxVar {\EVar[\ell]} {\T[\ell]} \CtxSeq \CtxRestrict{\E[\ell]}]
    {\E[\ell]}
    {\TU[\ell]}
    {\Eff[\ell]}
    [\CSet[\ell]]
  \\
  \ell' \in L
}{
  \InferType
    {\ECase[\ell\in L] \E \ECaseBranchDef}
    {\TU[\ell']}
    {\Eff \EffJoin \Eff[1] \EffJoin \Eff[2]}
    [\CSet \CSJoin \bigcup_{\ell\in L} \CSet[\ell] \CSJoin \left\{\TU[k] \CEq \TU[k'] \mid k,k' \in L \right\}]
}
