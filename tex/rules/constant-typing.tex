% Constant Typing

\def\RuleCmdPrefix{RuleCT}
\def\RuleNamePrefix{CT-}

% CT-Unit
\DefRule {Unit} {} {
  \HasCType\CUnit\BUnit
}

% CT-Fork
\DefRule {Fork} {} {
  \HasCType\CFork{
    \TArr{\MobMobile}{\DirNone,\EffPure}{
      (\TArr{\MobMobile}{\DirUnord,\EffImpure}\BUnit\BUnit)
    }{\BUnit}
  }
}

% CT-New
\DefRule {New} {
  \SNew \S
} {
  \HasCType
    {\CNew_{\S}}
    {\TArr{\MobMobile}{\DirNone,\EffPure}
      {\BUnit}
      {\TPairUnord
        {(\SSeq{\SAcq}{\SSeq\S\STerm})}
        {(\SSeq{\SAcq}{\SSeq{\SDual\S}{\SWait}})}
      }
    }
}

% CT-Drop
\DefRule {Drop} {} {
  \HasCType
  {\CDrop}{\TArr{\MobMobile}{\DirNone,\EffImpure}{\SDrop}{\BUnit}}
}

% CT-Acquire
\DefRule {Acquire} {} {
  \HasCType
  {\CAcq}{\TArr{\MobMobile}{\DirNone,\EffPure}{\SSeq\SAcq\S}{\S}}
}

% CT-LSplit
\DefRule {LSplit} {
  \neg (\S[2] \SEq \SSkip)
} {
  \HasCType
    {\CSplit_{\S[1],\S[2]}}
    {\TArr{\MobMobile}{\DirUnord,\EffPure}{\SSeq {\S[1]}{\S[2]}}{\TPairOrd{\S[1]}{\S[2]}}}
}

% CT-RSplit
\DefRule {RSplit} {
  \neg (\S[2] \SEq \SSkip)
} {
  \HasCType
  {\CSSplit_{\S[1],\S[2]}}
  {\TArr{\MobMobile}{\DirUnord,
    \EffPure}
    {\SSeq {\S[1]}{\S[2]}}
    {\TPairUnord{(\SSeq{\S[1]}{\SDrop})}{(\SSeq{\SAcq}{\S[2]})}}}
}

% CT-Wait
\DefRule {Wait} {} {
  \HasCType\CWait{\TArr{\MobMobile}{\DirNone,\EffImpure}{\SWait}\BUnit}
}

% CT-Close
\DefRule* [lab=CT-Close] {Term} {} {
  \HasCType\CTerm{\TArr{\MobMobile}{\DirNone,\EffImpure}{\STerm}\BUnit}
}

% CT-Send
\DefRule {Send} {
  \Mobile \T
}{
  \HasCType
    {\CSend_{\T}}
    {\TArr{\MobMobile}{\DirNone,\EffImpure}{\TPairUnord{\T}{\SSend\T}}\BUnit}
}

% CT-Recv
\DefRule {Recv} {
  \Mobile \T
}{
  \HasCType
    {\CRecv_{\T}}
    {\TArr{\MobMobile}{\DirNone,\EffImpure}{\SRecv\T}\T}
}

% CT-Select
\DefRule {Select} {
  k \in L
} {
  \HasCType{\CSelect{k}}{
    \TArr{\MobMobile}{\DirNone,\EffImpure}
      {\SChoice{\ell:\S[\ell]}{\ell\in L}}
      {\S[k]}
  }
}

% CT-Branch
\DefRule {Branch} {} {
  \HasCType\CBranch{
    \TArr{\MobMobile}{\DirNone,\EffImpure}
      {\SBranch{\ell:\S[\ell]}{\ell\in L}}
      {\TVariant{\ell:\S[\ell]}}
      %{\Sigma_k \S[k]}
  }
}

% CT-Discard
\DefRule {Discard} {} {
  \HasCType\CDiscard{
    \TArr{\MobMobile}{\DirNone,\EffImpure}
      {\SSkip}
      {\BUnit}
  }
}
