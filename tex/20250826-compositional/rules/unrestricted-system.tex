%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Typing rules for expressions

\def\RuleCmdPrefix{RuleTU}
\def\RuleNamePrefix{TU-}

\DefRule {Const} {
  \HasCType \TC \TT
}{
  \HasType \TCtx \TC \TT
}

\DefRule {Var} {
  \TCtxVar \TEVar \TT \in \TCtx
}{
  \HasType \TCtx \TEVar \TT
}

\DefRule {Abs} {
  \HasType {\TCtxExt{\TCtx}{\TCtxVar\TEVar{\TT[1]}}}
    {\TE}
    {\TT[2]}
}{
  \HasType \TCtx
    {\TEAbs \TEVar \TE}
    {\TTArr{\TT[1]}{\TT[2]}}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Typing rules for constants

\def\RuleCmdPrefix{RuleCU}
\def\RuleNamePrefix{CU-}

\DefRule {Unit} {} {
  \HasCType \TCUnit
    {\TBUnit}
}

\DefRule {New} {} {
  \HasCType \TCNew
    {\TTArr \TBUnit {\TTProd\TBChan\TBChan}}
}

\DefRule {Fork} {} {
  \HasCType \TCFork
    {\TTArr{(\TTArr \TBUnit \TBUnit)}{\TBUnit}}
}

\DefRule {Close} {} {
  \HasCType \TCTerm
    {\TTArr \TBChan \TBUnit}
}

\DefRule {Wait} {} {
  \HasCType \TCWait
    {\TTArr \TBChan \TBUnit}
}

\DefRule {Send} {} {
  \HasCType {{\TCSend}_{\TT}}
    {\TTArr {\TTProd \TT \TBChan} \TBUnit}
}

\DefRule {Recv} {} {
  \HasCType {{\TCRecv}_{\TT}}
    {\TTArr \TBChan \TT}
}

\DefRule {LSplit} {} {
  \HasCType {\TCSplit}
    {\TTArr \TBChan {\TTProd \TBChan \TBChan}}
}

\DefRule {RSplit} {} {
  \HasCType {\TCSSplit}
    {\TTArr \TBChan {\TTProd \TBChan \TBChan}}
}

\DefRule {Drop} {} {
  \HasCType {\TCDrop}
    {\TTArr \TBChan \TBUnit}
}

\DefRule {Acquire} {} {
  \HasCType {\TCAcq}
    {\TTArr \TBChan \TBChan}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Typing rules for processes

\def\RuleCmdPrefix{RuleTProc}
\def\RuleNamePrefix{TU-Proc-}

\DefRule {Expr} {
  \HasType \TCtx \TE \TBUnit
} {
  \ProcHasType \TCtx {\TPExp\TE}
}

\DefRule {Par} {
  \ProcHasType \TCtx {\TProc[1]} \\
  \ProcHasType \TCtx {\TProc[2]}
} {
  \ProcHasType \TCtx {\TPPar{\TProc[1]}{\TProc[2]}}
}

\DefRule {Res} {
  \ProcHasType {\TCtxExt{\TCtxExt \TCtx {\TCtxVar x \TBChanD}}{\TCtxVar y \TBChanD}} \TProc \\
} {
  \ProcHasType \TCtx {\TPNew{\TProc}}
}

\DefRule {Sync} {
  \ProcHasType {\TCtxExt \TCtx {\TCtxVar x \TBChanF}} \TProc \\
} {
  \ProcHasType \TCtx {\TPSync x{\TProc}}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Reduction rules

\def\RuleCmdPrefix{RuleUProcRed}
\def\RuleNamePrefix{RU-}

\DefRule {Exp} {
  \TE[1]  \Reduces \TE[2]
}{
  \TFCapp{\TE[1]} \Reduces \TFCapp{\TE[2]}
}

\DefRule {Fork} {} {
  \TFCapp{\TEApp \TCFork \TV}
  \Reduces
  \TPPar{\TFCapp\TCUnit}{\TPExp{\TEApp\TV\TCUnit}}
}

\DefRule {New} {} {
  \TFCapp{\TEApp \TCNew \TCUnit}
  \Reduces
  \TPNew {
    \TPSync* {z_1} {\TPSync* {z_2} {
      \TFCapp{\TEPair
        {(\TEChan [z_1]x)}
        {(\TEChan [z_2]y)}
      }
    }}
  }
}

\DefRule {LSplit} {} {
    \TPNew {(
      \TPPar {\TFCapp{\TEApp \TCSplit {(\TEChan [e_1]x[e_2])}}} \TProc
    )}
    \Reduces
    \TPNew {(
      \TPPar {\TFCapp{\TEPair
        {(\TEChan [e_1] x   )}
        {(\TEChan       x [e_2] )}
      }} \TProc
    )}
}

\DefRule {RSplit} {} {
    \TPNew {(
      \TPPar {\TFCapp{\TEApp \TCSSplit {(\TEChan [e_1]x[e_2])}}} \TProc
    )}
    \Reduces
    \TPNew {
      \TPSync z {(
        \TPPar {\TFCapp{\TEPair
          {(\TEChan [e_1] x [z]   )}
          {(\TEChan [z]   x [e_2] )}
        }} \TProc
      )} 
    }
}

\DefRule {Drop} {} {
    \TPNew {
      \TPSync z {(
        \TPPar {\TFCapp{\TEApp \TCDrop {(\TEChan x[z])}}} \TProc
      )} 
    }
    \Reduces 
    \TPNew {
      \TPSync* z {(
        \TPPar {\TFCapp{\TCUnit}} \TProc
      )} 
    }
}

\DefRule {Acquire} {} {
    \TPNew {
      \TPSync* z {(
        \TPPar {\TFCapp{\TEApp \TCAcq {(\TEChan [z]x[e])}}} \TProc
      )} 
    }
    \Reduces 
    \TPNew {(
      \TPPar {\TFCapp{\TEChan x[e]}} \TProc
    )}
}

\DefRule {Close} {} {
    \TPNew {(
      \TPPar {\TFCapp[1]{\TEApp\TCTerm {(\TEChan x)}}}
             {\TFCapp[2]{\TEApp\TCWait {(\TEChan y)}}}
    )}
    \Reduces 
    \TPPar {\TFCapp[1]{\TCUnit}} {\TFCapp[2]{\TCUnit}}
}

\DefRule {Com} {} {
  \TPNew {(
    \TPPar {\TFCapp[1]{\TEApp {\TCSend} {(\TEPair \TV {(\TEChan x[e_1])}}}} {
    \TPPar {\TFCapp[2]{\TEApp {\TCRecv} {(\TEChan y[e_2])}}} {\TProc}}
    % \TPPar {\TFCapp[1]{\TEApp {\TCSend_{\TT}} {(\TEPair \TV {(\TEChan x[e_1])}}}} {
    % \TPPar {\TFCapp[2]{\TEApp {\TCRecv_{\TT}} {(\TEChan y[e_2])}}} {\TProc}}
  )}
  \Reduces
  \TPNew {(
    \TPPar {\TFCapp[1]{\TCUnit}} {
    \TPPar {\TFCapp[2]{\TV}} {\TProc}}
  )}
}
