\def\RuleNamePrefix{CS-}
\def\RuleCmdPrefix{RuleCS}

% CS-Pair
\DefRule {Pair} {} {
  \AT \TPair \ATU \CEq \AT* \TPair \ATU*
  \csred
  \left\{
    \AT \CEq \AT*\!,
    \ATU \CEq \ATU*
  \right\}
}

\DefRule {Sum} {} {
  \TSum {\AT} {\ATU} \CEq \TSum {\AT*} {\ATU*}
  \csred
  \left\{
    \AT \CEq \AT*\!,
    \ATU \CEq \ATU*
  \right\}
}

\DefRule {Unit} {} {
  \BUnit \CEq \BUnit
  \csred
  \emptyset
}

\DefRule {Arrow} {} {
  \TArr{\Mob}{\Dir,\Eff}{\AT}{\ATU}
  \CEq
  \TArr{\Mob}{\Dir,\Eff}{\AT*}{\ATU*}
  \csred
  \{
    \AT* \CEq \AT\!,
    \ATU \CEq \ATU*
  \}
}

\DefRule {Context} {
  \AT \CEq \ATU \csred \CSet*
}{
  \CSet \cup \{ \AT\CEq\ATU \}
  \csred
  \CSet \cup \CSet*
}

\DefRule {Swap} {\AT \notin \TVAR} {
  \AT \CEq \TVar
  \csred
  \left\{
    \TVar \CEq \AT
  \right\}
}
