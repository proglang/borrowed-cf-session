% Reduction

\def\RuleCmdPrefix{RuleExpRed}
\def\RuleNamePrefix{E-}

% E-App
\DefRule {App} {} {
  \EApp{(\EAbs \EVar\E)}\V \EReduce \Subs \E\V\EVar
}

% E-Seq
\DefRule {Seq} {} {
  \ESeq \V \E \EReduce  \E
}

% E-PairElim
\DefRule {PairElim} {} {
  \ELet {\EPair xy} {\EPair uv} \E \EReduce \Subs {\Subs \E vx} uy
}

% E-Unfold
\DefRule {Unfold} {} {
  \EMu e \EReduce \Subs e {\EMu e} f
}

% E-VariantElim
\DefRule {VariantElim} {
  k\in L
} {
  \ECase [\ell\in L] {(\EInj k \V)} \ECaseBranchDef
    \EReduce \Subs {\E[k]} \V {\EVar[k]}
}

% E-Ctx
\DefRule {Ctx} {
  \E[1] \EReduce \E[2]
}{
  \termcolor{\EC{[\E[1]]}} \EReduce \termcolor{\EC{[\E[2]]}}
}

\def\RuleCmdPrefix{RuleProcRed}
\def\RuleNamePrefix{R-}

% R-Exp
\DefRule {Exp} {
  \E[1] \EReduce \E[2]
}{
  \FCapp {\E[1]} \PReduce \FCapp{\E[2]}
}

% R-Discard
\DefRule {Discard} {} {
  \PNew {\BSeq{\EVar}{\BindGroup[1]}} {\BindGroup[2]} {(
    \PPar { \FCapp{\EApp\CDiscard\EVar} }
          { \Proc }
  )}
  \PReduce
  \PNew {\BindGroup[1]} {\BindGroup[2]} {(
    \PPar { \FCapp{\CUnit} }
          { \Proc }
  )}
}

% R-Close
\DefRule {Close} {} {
  \PNew xy {(\PPar { {\FCapp[1]{\EApp\CTerm x}}} { {\FCapp[2]{\EApp\CWait y}}})} 
  \PReduce
  \PPar { {\FCapp[1]{\CUnit}}} { {\FCapp[2]{\CUnit}}}
}

% R-RSplit
\DefRule {RSplit} {} {
  \PNew {\BindGroup[1]\EVar\BindGroup[2]}{\BindGroup[3]} {
    (\PPar{\FCapp{\EApp \CSSplit {x}}}\Proc)
  }
  \PReduce
  \PNew {\BindGroup[1]\EVar[1]\BSep\EVar[2]\BindGroup[2]}{\BindGroup[3]} {
    ({ \PPar{\FCapp{\EPair{\EVar[1]}{\EVar[2]}}}\Proc})
  }
}

% R-LSplit
\DefRule {LSplit} {} {
  \PNew {\BindGroup[1]\EVar\BindGroup[2]}{\BindGroup[3]} {
    (\PPar{\FCapp{\EApp \CSplit {x}}}\Proc)
  }
  \PReduce
  \PNew {\BindGroup[1]\EVar[1]\EVar[2]\BindGroup[2]}{\BindGroup[3]} {
    (\PPar{\FCapp{\EPairOrd{\EVar[1]}{\EVar[2]}}}\Proc)
  }
}

% R-Drop
\DefRule {Drop} {} {
  \PNew {\BWithSep{\EVar}{\BindGroup[1]}}{\BindGroup[2]}{ (\PPar{\FCapp{\EApp \CDrop {\EVar}}}\Proc)}
  \PReduce
  \PNew {\BLeadSep{\BindGroup[1]}}{\BindGroup[2]} { ({ \PPar{\FCapp{\CUnit}}\Proc})}
}

% R-Acquire
\DefRule {Acquire} {} {
  \PNew {\BLeadSep{\BSeq{\EVar}{\BindGroup[1]}}}{\BindGroup[2]}{ (\PPar{\FCapp{\EApp \CAcq{\EVar}}}\Proc)}
  \PReduce
  \PNew {\BSeq{\EVar}{\BindGroup[1]}}{\BindGroup[2]} { ({ \PPar{\FCapp{\EVar}}\Proc})}
}


% R-Com
\DefRule {Com} {} {
  \PNew {\BSeq x{\BindGroup[1]}}{\BSeq y {\BindGroup[2]}}{
    (\PPar{\PPar { {\FCapp[1]{\EApp \CSend {(\EPair\V x)}}}} {
        {\FCapp[2]{\EApp\CRecv y}}}} \Proc)} 
  \PReduce
  \PNew {\BindGroup[1]}{\BindGroup[2]}
  {(\PPar{\PPar{ {\FCapp[1]{\CUnit}}} { {\FCapp[2]{\V}}}} \Proc)}
}

% R-Choice
\DefRule {Choice} {} {
  \PNew {\BSeq x{\BindGroup[1]}}{\BSeq y {\BindGroup[2]}} {(
    \PPar {\FCapp[1]{\EApp{\CSelect{k}} x}} {
      \PPar {\FCapp[2]{ \EApp\CBranch y }}
        \Proc
    }
  )}
  \PReduce
  \PNew {\BSeq x{\BindGroup[1]}}{\BSeq y {\BindGroup[2]}} {(
    \PPar {\FCapp[1]{x}} {
      \PPar {\FCapp[2]{\EInj k y}}
        \Proc
    }
  )}
}

% R-New
\DefRule {New} {} {
  { {\FCapp{\EApp \CNew \CUnit}}}
  \PReduce
  \PNew {\BLeadSep{x}}{\BLeadSep{y}} { {\FCapp{\EPair xy}}}
}

% R-Fork
\DefRule {Fork} {} {
  { {\FCapp{\EApp \CFork \V}}}
  \PReduce
  \PPar { {\FCapp{\CUnit}}} {\PExp {\EApp\V\CUnit}}
}

% R-Struct
\DefRule {Struct} {
  \Proc[1] \equiv \Proc*[1] \\
  \Proc*[1] \PReduce \Proc*[2] \\
  \Proc*[2] \equiv \Proc[2]
}{
  \Proc[1] \PReduce \Proc[2]
}

% R-Par
\DefRule {Par} {
  \Proc[1] \PReduce \Proc[2]
}{
  \PPar {\Proc[1]}{\ProcQ} \PReduce \PPar {\Proc[2]}{\ProcQ}
}

% R-Bind
\DefRule {Bind} {
  \Proc[1] \PReduce \Proc[2]
}{
  \PNew{\BindGroup[1]}{\BindGroup[2]} {\Proc[1]} \PReduce \PNew {\BindGroup[1]}{\BindGroup[2]}{\Proc[2]}
}
