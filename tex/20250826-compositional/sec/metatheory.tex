\subsection{Type Safety for Declarative Typing}

\begin{theorem}[Expression progress]
  Let $\Ctx$ only contain session types.
  If $\HasType \Ctx \E \T$, then either $\E$ is a value or $\E$ is blocked on concurrent actions (\mw{ref appendix}) or there exists an $\E*$ such that $\E \EReduce \E*$.
\end{theorem}
\mw{deadlocks are possible.}

\begin{theorem}[Expression preservation]
  Let $\Ctx$ only contain session types.
  If $\HasType \Ctx \E \T$ and $\E \EReduce \E*$, then
  $\HasType \Ctx {\E*} \T$.
\end{theorem}

\begin{theorem}[Process progress]
  Let $\Ctx$ only contain session types.
  If $\ProcHasType \Ctx \Proc$, then either $\Proc \Congruent \PExp{\CUnit}$ or an expression in $\Proc$ is blocked on concurrent actions (\mw{ref appendix}) or
    there exists an $\Proc*$ such that $\Proc \PReduce \Proc*$.
\end{theorem}

\begin{theorem}[Process preservation]
  Let $\Ctx$ only contain session types.
  If $\ProcHasType \Ctx \Proc$ and $\Proc \PReduce \Proc*$, then
  $\ProcHasType \Ctx {\Proc*}$.
\end{theorem}

\subsection{Weak Operational Correspondence for Translation}

\begin{theorem}[Forward simulation]
  Let $\Ctx$ only contain session types and $\Sub$ be a value substitution.
  Additionally, let $\Proc \TransProc[\Sub] \TProc*$ and $\ProcQ \TransProc[\Sub] \TProcQ*$.
  If $\ProcHasType \Ctx \Proc$ and $\Proc \PReduce \ProcQ$, then 
  either $\TProc* \TPReduce^{*} \TProcQ*$ or $\TProc* \Congruent \TProcQ*$.
\end{theorem}

\begin{theorem}[Backward simulation]
  Let $\Ctx$ only contain session types and $\Sub$ be a value substitution.
  Additionally, let $\Proc \TransProc[\Sub] \TProc*$.
  If $\ProcHasType \Ctx \Proc$ and $\TProc* \TPReduce \TProcR*$, then there
  exist $\ProcQ$ and $\TProcQ*$ such that $\Proc \PReduce^{*} \ProcQ$ and
  $\TProcR* \TPReduce^{*} \TProcQ*$ where $\ProcQ \TransProc[\Sub] \TProcQ*$.
\end{theorem}

\subsection{Algorithmic Typing}

\begin{theorem}[Soundness (checking)]
  If $\CheckType \E \AT \Eff$ and $\sigma \models \CSet$ is a closing
  substitution such that $\Fv(\AT) \subseteq \dom(\sigma)$, then
  $\HasType \Ctx \E {\AT{[\sigma]}}$.
\end{theorem}

\begin{theorem}[Soundness (inference)]
  If $\InferType \E \AT \Eff$ and $\sigma \models \CSet$ is a closing
  substitution such that $\Fv(\AT) \subseteq \dom(\sigma)$, then
  $\HasType \Ctx \E {\AT{[\sigma]}}$.
\end{theorem}

\begin{theorem}[Completeness (checking)]
  If $\HasType \Ctx \E \T$, then
  $\CheckType \E \AT \Eff$ for some type $\AT$, set of constraints $\CSet$,
  and closing substitution $\sigma$ such that $\T = \AT{[\sigma]}$.
\end{theorem}

\begin{theorem}[Completeness (inference)]
  If $\HasType \Ctx \E \T$ and $\E$ is in synthesis form, then
  $\InferType \E \AT \Eff$ for some type $\AT$, set of constraints $\CSet$,
  and closing substitution $\sigma$ such that $\T = \AT{[\sigma]}$.
\end{theorem}
