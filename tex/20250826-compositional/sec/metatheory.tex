\subsection{Structural Properties}

\begin{lemma}[Unrestricted decomposition]
  Every context $\Ctx$ can be separated into contexts $\Ctx[1], \Ctx[2]$ with
  $\Ctx = \Ctx[1] \CtxPar \Ctx[2]$ such that $\Unr\Ctx[1]$ and $\Unr\T$ is
  not derivable for any $\CtxVar\EVar\T \in \Ctx[2]$.
\end{lemma}
\begin{proof}
  \texttt{unjoinUnr}
\end{proof}

\begin{lemma}[Monoidal properties of contexts]
  The context constructors satisfy the following properties with respect to
  context equality.
  \begin{itemize}
    \item Sequential composition \textup{($\Ctx[1] \CtxSeq \Ctx[2]$)} forms a
      monoid with identity $\CtxEmpty$.
    \item Parallel composition \textup{($\Ctx[1] \CtxPar \Ctx[2]$)} forms a
      commutative monoid with identity $\CtxEmpty$.
  \end{itemize}
\end{lemma}

\begin{lemma}[Unrestricted contexts are mobile]
  $\Unr\Ctx \implies \Mobile\Ctx$.
\end{lemma}
\begin{proof}
  By induction over $\Ctx$ and the types of bound variables therein.
\end{proof}

\subsection{Translation}

\begin{theorem}[Forward simulation]
  Assume $\mathsf{VSub}(\Sub)$ and $\mathsf{ChanCx}(\Ctx)$.
  If $\ProcHasType \Ctx \Proc$, $\Proc \TransProc[\Sub] \TProc*$,
  $\ProcQ \TransProc[\Sub] \TProcQ*$, and $\Proc \PReduce \ProcQ$, then
  \[
    \bigl(\TProc* \TPReduce^{*} \TProcQ*\bigr)
    \;\lor\;
    \bigl(\TProc* \Congruent \TProcQ*\bigr).
  \]
\end{theorem}

\begin{theorem}[Backward simulation]
  Assume $\mathsf{VSub}(\Sub)$ and $\mathsf{ChanCx}(\Ctx)$.
  If $\ProcHasType \Ctx \Proc$, $\Proc \TransProc[\Sub] \TProc*$,
  $\TProc[1] \approx \TProc*$, and $\TProc[1] \TPReduce \TProc[2]$, then there
  exist $\ProcQ$ and $\TProc[3]$ such that
  \[
    \Proc \PReduce^{*} \ProcQ
    \;\land\;
    \TProc[2] \TPReduce^{?} \TProc[3]
    \;\land\;
    \TProc[3] \approx \TProcQ*.
  \]
  where $\ProcQ \TransProc[\Sub] \TProcQ*$.
\end{theorem}

\subsection{Algorithmic Typing}

\begin{lemma}[Soundness (checking)]
  If $\CheckType \E \AT \Eff$ and $\sigma \models \CSet$ is a closing
  substitution such that $\Fv(\AT) \subseteq \dom(\sigma)$, then
  $\HasType \Ctx \E {\AT{[\sigma]}}$.
\end{lemma}

\begin{lemma}[Soundness (inference)]
  If $\InferType \E \AT \Eff$ and $\sigma \models \CSet$ is a closing
  substitution such that $\Fv(\AT) \subseteq \dom(\sigma)$, then
  $\HasType \Ctx \E {\AT{[\sigma]}}$.
\end{lemma}

\begin{lemma}[Completeness (checking)]
  If $\HasType \Ctx \E \T$, then
  $\CheckType \E \AT \Eff$ for some type $\AT$, set of constraints $\CSet$,
  and closing substitution $\sigma$ such that $\T = \AT{[\sigma]}$.
\end{lemma}

\begin{lemma}[Completeness (inference)]
  If $\HasType \Ctx \E \T$ and $\E$ is in synthesis form, then
  $\InferType \E \AT \Eff$ for some type $\AT$, set of constraints $\CSet$,
  and closing substitution $\sigma$ such that $\T = \AT{[\sigma]}$.
\end{lemma}
