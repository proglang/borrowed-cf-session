% Translation of expressions

\def\RuleCmdPrefix{RuleTransExp}
\def\RuleNamePrefix{Tr-Exp-}

\DefRule {SubVar} {
  \EVar \in \Sub
} {
  \HasType {\Ctx} \EVar \T \TransExp \Sub(\EVar)
}

\DefRule {CtxVar} {
  \EVar \notin \Sub
} {
  \HasType {\Ctx} \EVar \T \TransExp \EVar
}

\DefRule {AbsLin} {
  \HasType {\CtxExtUnord\Ctx{\CtxVar\EVar{\T[1]}}} \E {\T[2]}
    \TransExp \TE*
} {
  \HasType \Ctx {(\EAbs\EVar\E)} {(\TArr{\Mob}{\DirUnord,\Eff}{\T[1]}{\T[2]})}
    \TransExp \TEAbs\TEVar{\TE*}
}

\DefRule {Send} {} {
  \HasType {\Ctx} \CSend
    {\TArr \MobMobile {\DirUnord,\EffImpure} {\T\PairUnord{\SSend\T}} \BUnit}
  \TransExp
  \TCSend_{\TrT\T}
}

\DefRule {Recv} {} {
  \HasType {\Ctx} \CRecv
    {\TArr \MobMobile {\DirUnord,\EffImpure} {\SRecv\T} \T}
  \TransExp
  \TCRecv_{\TrT\T}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Translation of processes

\def\RuleCmdPrefix{RuleTransProc}
\def\RuleNamePrefix{Tr-}

% Tr-Exp
\DefRule {Exp} {} {
  \PExp{\E}
  \TransProc
  \TPExp{\metacolor{\Sub{[\E*]}}}
}

% Tr-Par
\DefRule {Par} {
  \Proc[1] \TransProc \TProc*[1] \\
  \Proc[2] \TransProc \TProc*[2]
} {
  \PPar {\Proc[1]} {\Proc[2]}
  \TransProc
  \TPPar {\TProc*[1]} {\TProc*[2]}
}

% Tr-Bind
\DefRule {Bind} {
  \BindGroup[1] \TransBind[\TEChan \TEVar] {\TCC[1]} {\Sub[1]} \\
  \BindGroup[2] \TransBind[\TEChan \TEVary] {\TCC[2]} {\Sub[2]} \\
  \Proc \TransProc[{\Sub \cup \Sub[1] \cup \Sub[2]}] \TProc* \\
} {
  \PNew {\BindGroup[1]}{\BindGroup[2]} \Proc
  \TransProc
  \TPNew {\TCC[1][\TCC[2][\TProc*]]}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Translation of bind groups

\def\RuleCmdPrefix{RuleTransBind}
\def\RuleNamePrefix{Tr-Bind}

% Tr-BindVar
\DefRule {Var} {} {
  \procbindcolor{\EVar} \TransBind \TCCEmpty {\SubSingle \EVar \E}
}

% Tr-BindNil
\DefRule {Nil} {} {
  \BEmp \TransBind \TCCEmpty \emptyset
}

% Tr-BindSeq
\DefRule {Seq} {
  \Bind[1] \TransBind[{\TEChan[\E[1]] \TEVar}] {\TCC[1]} {\Sub[1]} \\
  \Bind[2] \TransBind[{\TEChan \TEVar [\E[2]]}] {\TCC[2]} {\Sub[2]} \\
}{
  \BSeq {\Bind[1]} {\Bind[2]} \TransBind[{\TEChan[\E[1]] \TEVar [\E[2]]}]
    \TCCEmpty {\Sub[1] \cup \Sub[2]}
}

% Tr-BindPar
\DefRule {Par} {
  \BindGroup[1] \TransBind[{\TEChan[\E[1]] \TEVar [\TEVarz]}] {\TCC[1]} {\Sub[1]} \\
  \BindGroup[2] \TransBind[{\TEChan[\TEVarz] \TEVar [\E[1]]}] {\TCC[2]} {\Sub[2]} \\
  \BindFlag{\BindGroup[1]} = \Flag
}{
  \BPar {\BindGroup[1]} {\BindGroup[2]} \TransBind[{\TEChan[\E[1]] \TEVar [\E[2]]}]
    {\TPSync {z} {(\TPPar {\TPFlag z \Flag} {\TCC[1][\TCC[2]]})}}
    {\Sub[1] \cup \Sub[2]}
}
