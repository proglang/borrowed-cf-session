%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Typing rules for expressions

\def\RuleCmdPrefix{RuleTU}
\def\RuleNamePrefix{TU-}

\DefRule {Const} {
  \HasCType \TC \TT
}{
  \HasType \TCtx \TC \TT
}

\DefRule {Var} {
  \TCtxVar \TEVar \TT \in \TCtx
}{
  \HasType \TCtx \TEVar \TT
}

\DefRule {Abs} {
  \HasType {\TCtxExt{\TCtx}{\TCtxVar\TEVar{\TT[1]}}}
    {\TE}
    {\TT[2]}
}{
  \HasType \TCtx
    {\TEAbs \TEVar \TE}
    {\TTArr{\TT[1]}{\TT[2]}}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Typing rules for constants

\def\RuleCmdPrefix{RuleCU}
\def\RuleNamePrefix{CU-}

\DefRule {Unit} {} {
  \HasCType \TCUnit
    {\TBUnit}
}

\DefRule {New} {} {
  \HasCType \TCNew
    {\TTArr \TBUnit {\TTProd\TBChan\TBChan}}
}

\DefRule {Fork} {} {
  \HasCType \TCFork
    {\TTArr{(\TTArr \TBUnit \TBUnit)}{\TBUnit}}
}

\DefRule {Close} {} {
  \HasCType \TCTerm
    {\TTArr \TBChan \TBUnit}
}

\DefRule {Wait} {} {
  \HasCType \TCWait
    {\TTArr \TBChan \TBUnit}
}

\DefRule {Send} {} {
  \HasCType {{\TCSend}_{\TT}}
    {\TTArr {\TTProd \TT \TBChan} \TBUnit}
}

\DefRule {Recv} {} {
  \HasCType {{\TCRecv}_{\TT}}
    {\TTArr \TBChan \TT}
}

\DefRule {LSplit} {} {
  \HasCType {\TCSplit}
    {\TTArr \TBChan {\TTProd \TBChan \TBChan}}
}

\DefRule {RSplit} {} {
  \HasCType {\TCSSplit}
    {\TTArr \TBChan {\TTProd \TBChan \TBChan}}
}

\DefRule {Drop} {} {
  \HasCType {\TCDrop}
    {\TTArr \TBChan \TBUnit}
}

\DefRule {Acquire} {} {
  \HasCType {\TCAcq}
    {\TTArr \TBChan \TBChan}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Typing rules for processes

\def\RuleCmdPrefix{RuleTProc}
\def\RuleNamePrefix{TU-Proc-}

\DefRule {Expr} {
  \HasType \TCtx \TE \TBUnit
} {
  \ProcHasType \TCtx {\TPExp\TE}
}

\DefRule {Par} {
  \ProcHasType \TCtx {\TProc[1]} \\
  \ProcHasType \TCtx {\TProc[2]}
} {
  \ProcHasType \TCtx {\TPPar{\TProc[1]}{\TProc[2]}}
}

\DefRule {Res} {
  \ProcHasType {\TCtxExt{\TCtxExt \TCtx {\TCtxVar x \TBChanD}}{\TCtxVar y \TBChanD}} \TProc \\
} {
  \ProcHasType \TCtx {\TPNew{\TProc}}
}

\DefRule {Sync} {
  \ProcHasType {\TCtxExt \TCtx {\TCtxVar x \TBChanF}} \TProc \\
} {
  \ProcHasType \TCtx {\TPSync x{\TProc}}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Reduction rules

\def\RuleCmdPrefix{RuleUProcRed}
\def\RuleNamePrefix{RU-}

\DefRule {Exp} {
  \E[1]  \Reduces \E[2]
}{
  \TFCapp{\E[1]} \Reduces \TFCapp{\E[2]}
}

\DefRule {Fork} {} {
  \TFCapp{\EApp \CFork \V}
  \Reduces
  \TPPar{\TFCapp\CUnit}{\TPExp{\EApp\V\CUnit}}
}

\DefRule {New} {} {
  \TFCapp{\EApp \CNew \CUnit}
  \Reduces
  \TPNew {
    \TPSync {z_1} {\TPSync {z_2} {(
      \TPPar{
        \TPPar {\TPFlag {z_1} \FlagSet}
               {\TPFlag {z_2} \FlagSet}
      }{
        \TFCapp{\EPair
          {(\TEChan [z_1]x)}
          {(\TEChan [z_2]y)}
        }
      }
    )}}
  }
}

\DefRule {LSplit} {} {
    \TPNew {(
      \TPPar {\TFCapp{\EApp \CSplit {(\TEChan [e_1]x[e_2])}}} \TProc
    )}
    \Reduces
    \TPNew {(
      \TPPar {\TFCapp{\EPair
        {(\TEChan [e_1] x   )}
        {(\TEChan       x [e_2] )}
      }} \TProc
    )}
}

\DefRule {RSplit} {} {
    \TPNew {(
      \TPPar {\TFCapp{\EApp \CSSplit {(\TEChan [e_1]x[e_2])}}} \TProc
    )}
    \Reduces
    \TPNew {
      \TPSync z {(
        \TPPar {\TPFlag z \FlagUnset} {
          \TPPar {\TFCapp{\EPair
            {(\TEChan [e_1] x [z]   )}
            {(\TEChan [z]   x [e_2] )}
          }} \TProc
        }
      )} 
    }
}

\DefRule {Drop} {} {
    \TPNew {(
      \TPPar {\TPFlag z \FlagUnset} {
        \TPPar {\TFCapp{\EApp \CDrop {(\TEChan x[z])}}} \TProc
      }
    )}
    \Reduces 
    \TPNew {(
      \TPPar {\TPFlag z \FlagSet} {
        \TPPar {\TFCapp{\CUnit}} \TProc
      }
    )}
}

\DefRule {Acquire} {} {
    \TPNew {
      \TPSync z {(
        \TPPar {\TPFlag z \FlagSet} {
          \TPPar {\TFCapp{\EApp \CAcq {(\TEChan [z]x[e])}}} \TProc
        }
      )}
    }
    \Reduces 
    \TPNew {(
      \TPPar {\TFCapp{\TEChan x[e]}} \TProc
    )}
}

\DefRule {Close} {} {
    \TPNew {(
      \TPPar {\TFCapp[1]{\EApp\CTerm {(\TEChan x)}}}
             {\TFCapp[2]{\EApp\CWait {(\TEChan y)}}}
    )}
    \Reduces 
    \TPPar {\TFCapp[1]{\CUnit}} {\TFCapp[2]{\CUnit}}
}

\DefRule {Com} {} {
  \TPNew {(
    \TPPar {\TFCapp[1]{\EApp {\CSend} {(\EPair \TV {(\TEChan x[e_1])}}}} {
    \TPPar {\TFCapp[2]{\EApp {\CRecv} {(\TEChan y[e_2])}}} {\TProc}}
  )}
  \Reduces
  \TPNew {(
    \TPPar {\TFCapp[1]{\CUnit}} {
    \TPPar {\TFCapp[2]{\TV}} {\TProc}}
  )}
}
