% Translation of processes

\def\RuleCmdPrefix{RuleTransProc}
\def\RuleNamePrefix{Tr-}

% Tr-Exp
\DefRule {Exp} {} {
  \PExp{\E}
  \TransProc
  \TPExp{\metacolor{\Sub{[\E*]}}}
}

% Tr-Par
\DefRule {Par} {
  \Proc[1] \TransProc \TProc*[1] \\
  \Proc[2] \TransProc \TProc*[2]
} {
  \PPar {\Proc[1]} {\Proc[2]}
  \TransProc
  \TPPar {\TProc*[1]} {\TProc*[2]}
}

% Tr-Bind
\DefRule {Bind} {
  \BindGroup[1] \TransBind[\TEChan \TEVar] {\TCC[1]}  [\Sub[1]] \\
  \BindGroup[2] \TransBind[\TEChan \TEVary] {\TCC[2]} [\Sub[2]] \\
  \Proc \TransProc[{\Sub \cup \Sub[1] \cup \Sub[2]}] \TProc* \\
} {
  \PNew {\BindGroup[1]}{\BindGroup[2]} \Proc
  \TransProc
  \TPNew {\TCC[1][\TCC[2][\TProc*]]}
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Translation of binder groups

\def\RuleCmdPrefix{RuleTransBind}
\def\RuleNamePrefix{Tr-Bind}

% Tr-BindSep
\DefRule {Sep} {
  \BindGroup[1] \TransBind[{\TEChan[\E[1]] \TEVar [\TEVarz]}] {\TCC[1]} [\Sub[1]] \\
  \BindGroup[2] \TransBind[{\TEChan[\TEVarz] \TEVar [\E[2]]}] {\TCC[2]} [\Sub[2]] \\
  \BindFlag{\BindGroup[1]} = \Flag
}{
  \BWithSep{\BindGroup[1]}{\BindGroup[2]} \TransBind[{\TEChan[\E[1]] \TEVar [\E[2]]}]
    {\TPSync \Flag \targetproccolor{(\TCC[1][\TCC[2]])}}
    [\Sub[1] \cup \Sub[2]]
}

% Tr-BindSeq
\DefRule {Seq} {
  \BindGroup[1] \TransBind[\TEChan[\E[1]]{\TEVar}] {\TCC[1]} [\Sub[1]] \\
  \BindGroup[2] \TransBind[\TEChan{\TEVar}[\E[2]]] {\TCC[2]} [\Sub[2]] \\
}{
  \BSeq{\BindGroup[1]}{\BindGroup[2]} \TransBind[\TEChan[\E[1]]{\TEVar}[\E[2]]]
    {\targetproccolor{(\TCC[1][\TCC[2]])}}
    [\Sub[1] \cup \Sub[2]]
}

% Tr-BindVar
\DefRule {Var} {} {
  \EVar \TransBind \TCCEmpty [\{ \EVar\mapsto\E \}]
}

% Tr-BindNil
\DefRule {Nil} {} {
  \BEmp \TransBind \TCCEmpty [\emptyset]
}
